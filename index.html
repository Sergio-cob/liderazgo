// ─── AUTH STATE ───────────────────────────────────────────

let accessToken = null;

let allData = [];


function signIn() {

  const client = google.accounts.oauth2.initTokenClient({

    client_id: CLIENT_ID,

    scope: SCOPES,

    callback: (response) => {

      if (response.error) { showToast('Error al iniciar sesión', true); return; }

      accessToken = response.access_token;

      onSignedIn();

    }

  });

  client.requestAccessToken();

}


function onSignedIn() {

  document.getElementById('loginScreen').style.display = 'none';

  document.getElementById('mainApp').style.display = 'block';

  document.getElementById('authArea').innerHTML = `

    <div class="user-pill">✓ Conectado a Google Sheets</div>

    <button class="btn-signout" onclick="signOut()">Salir</button>

  `;

  ensureHeaders();

  buildForm();

  document.getElementById('evalDate').value = new Date().toISOString().split('T')[0];

}


function signOut() {

  accessToken = null;

  document.getElementById('loginScreen').style.display = 'flex';

  document.getElementById('mainApp').style.display = 'none';

  document.getElementById('authArea').innerHTML = `

    <button class="google-btn" onclick="signIn()" style="padding:8px 16px;font-size:0.8rem;width:auto;border-color:rgba(255,255,255,0.3);background:rgba(255,255,255,0.1);color:#f0ece4;">

      Iniciar sesión con Google

    </button>

  `;

}


// ─── SHEETS API ───────────────────────────────────────────

async function sheetsRequest(endpoint, method = 'GET', body = null) {

  const url = `https://sheets.googleapis.com/v4/spreadsheets/${SHEET_ID}${endpoint}`;

  const opts = {

    method,

    headers: {

      'Authorization': `Bearer ${accessToken}`,

      'Content-Type': 'application/json'

    }

  };

  if (body) opts.body = JSON.stringify(body);

  const res = await fetch(url, opts);

  if (!res.ok) throw new Error(`Sheets API error: ${res.status}`);

  return res.json();

}


async function ensureHeaders() {

  try {

    const data = await sheetsRequest(`/values/${SHEET_NAME}!A1:A1`);

    if (!data.values) {

      // Sheet is empty, write headers

      await sheetsRequest(`/values/${SHEET_NAME}!A1`, 'PUT', {

        range: `${SHEET_NAME}!A1`,

        majorDimension: 'ROWS',

        values: [HEADERS]

      });

    }

  } catch(e) {

    // Sheet tab might not exist, try to create it

    try {

      await sheetsRequest('', 'POST', { // batchUpdate

      });

    } catch(_) {}

    // Try writing headers anyway

    try {

      await sheetsRequest(`/values/${SHEET_NAME}!A1?valueInputOption=RAW`, 'PUT', {

        range: `${SHEET_NAME}!A1`,

        majorDimension: 'ROWS',

        values: [HEADERS]

      });

    } catch(_) {}

  }

}


async function appendRow(rowData) {

  await sheetsRequest(

    `/values/${SHEET_NAME}!A1:append?valueInputOption=RAW&insertDataOption=INSERT_ROWS`,

    'POST',

    { values: [rowData] }

  );

}


async function loadData() {

  const container = document.getElementById('tableContainer');

  container.innerHTML = `<div class="status-bar info"><span class="spinner"></span> Cargando datos desde Google Sheets...</div>`;

  try {

    const data = await sheetsRequest(`/values/${SHEET_NAME}`);

    const rows = data.values || [];

    if (rows.length <= 1) { allData = []; }

    else {

      const headers = rows[0];

      allData = rows.slice(1).map(row => {

        const obj = {};

        headers.forEach((h, i) => obj[h] = row[i] || '');

        return obj;

      });

    }

    renderTable();

    renderCompare();

  } catch(e) {

    container.innerHTML = `<div class="status-bar error">Error al cargar datos. Verifica que la hoja "Evaluaciones" existe en tu Google Sheets.</div>`;

  }

}

async function saveEvaluation() {

  const leaderName = document.getElementById('leaderName').value.trim();

  const evalDate = document.getElementById('evalDate').value;

  const role = document.getElementById('intervieweeRole').value.trim();

  const org = document.getElementById('orgName').value.trim();

  const notes = document.getElementById('notes').value.trim();


  if (!leaderName) { showToast('Ingresa el nombre o código del líder', true); return; }

  if (!evalDate) { showToast('Selecciona la fecha', true); return; }


  const answers = {};

  let allAnswered = true;

  DIMENSIONS.forEach(dim => {

    dim.questions.forEach((_, qi) => {

      const qid = `${dim.id}_${qi}`;

      const sel = document.querySelector(`.likert-btn.selected[data-qid="${qid}"]`);

      if (!sel) allAnswered = false;

      else answers[qid] = parseInt(sel.dataset.val);

    });

  });


  if (!allAnswered) { showToast('Responde todas las preguntas', true); return; }


  // Build row

  const dimScores = DIMENSIONS.map(dim => {

    const vals = dim.questions.map((_, qi) => answers[`${dim.id}_${qi}`]);

    return vals.reduce((a,b) => a+b, 0) / vals.length;

  });

  const overall = dimScores.reduce((a,b) => a+b, 0) / dimScores.length;


  const row = [

    evalDate, leaderName, org, role,

    ...DIMENSIONS.flatMap(dim => dim.questions.map((_, qi) => answers[`${dim.id}_${qi}`])),

    ...dimScores.map(s => s.toFixed(2)),

    overall.toFixed(2),

    notes

  ];


  const btn = document.getElementById('submitBtn');

  btn.disabled = true;

  btn.innerHTML = '<span class="spinner"></span> Guardando...';


  try {

    await appendRow(row);

    showToast('✓ Guardado en Google Sheets');

    resetForm();

  } catch(e) {

    showToast('Error al guardar. Intenta nuevamente.', true);

  } finally {

    btn.disabled = false;

    btn.innerHTML = 'Guardar en Google Sheets';

  }

}


function resetForm() {

  document.getElementById('leaderName').value = '';

  document.getElementById('intervieweeRole').value = '';

  document.getElementById('orgName').value = '';

  document.getElementById('notes').value = '';

  document.querySelectorAll('.likert-btn.selected').forEach(b => b.classList.remove('selected'));

  document.getElementById('evalDate').value = new Date().toISOString().split('T')[0];

}


// ─── TABLE ────────────────────────────────────────────────

function renderTable() {

  const filterLeader = document.getElementById('filterLeader').value;

  const leaders = [...new Set(allData.map(r => r['Líder']))].filter(Boolean).sort();


  const sel = document.getElementById('filterLeader');

  const prev = sel.value;

  sel.innerHTML = '<option value="">Todos los líderes</option>';

  leaders.forEach(l => {

    const opt = document.createElement('option');

    opt.value = l; opt.textContent = l;

    if (l === prev) opt.selected = true;

    sel.appendChild(opt);

  });


  const filtered = filterLeader ? allData.filter(r => r['Líder'] === filterLeader) : allData;

  const container = document.getElementById('tableContainer');


  if (!filtered.length) {

    container.innerHTML = `<div class="empty-state"><div class="icon">📋</div><p>No hay evaluaciones registradas aún.</p></div>`;

    return;

  }


  const sorted = [...filtered].sort((a,b) => new Date(b['Fecha']) - new Date(a['Fecha']));


  container.innerHTML = `

    <div class="table-wrap">

      <table class="records-table">

        <thead>

          <tr>

            <th>Fecha</th>

            <th>Líder</th>

            <th>Organización</th>

            <th>Rol</th>

            ${DIMENSIONS.map(d => `<th>${d.label}</th>`).join('')}

            <th>Global</th>

          </tr>

        </thead>

        <tbody>

          ${sorted.map(r => `

            <tr>

              <td>${r['Fecha']}</td>

              <td style="font-weight:500">${r['Líder']}</td>

              <td style="color:var(--muted)">${r['Organización'] || '—'}</td>

              <td style="color:var(--muted)">${r['Rol Entrevistado'] || '—'}</td>

              ${DIMENSIONS.map(d => `<td>${scorePill(parseFloat(r[`Promedio ${d.label}`]) || 0)}</td>`).join('')}

              <td>${scorePill(parseFloat(r['Promedio Global']) || 0)}</td>

            </tr>

          `).join('')}

        </tbody>

      </table>

    </div>

  `;

}


function scorePill(score) {

  const cls = score >= 4 ? 'score-high' : score >= 2.5 ? 'score-mid' : 'score-low';

  return `<span class="score-pill ${cls}">${score.toFixed(2)}</span>`;

}


// ─── COMPARE ──────────────────────────────────────────────

function renderCompare() {

  const leaders = [...new Set(allData.map(r => r['Líder']))].filter(Boolean).sort();

  const grid = document.getElementById('compareGrid');


  if (!leaders.length) {

    grid.innerHTML = `<div class="empty-state" style="grid-column:1/-1"><div class="icon">📊</div><p>Guarda evaluaciones para ver comparaciones.</p></div>`;

    return;

  }


  grid.innerHTML = leaders.map(leader => {

    const records = allData.filter(r => r['Líder'] === leader);

    const count = records.length;

    const overall = records.reduce((a,r) => a + (parseFloat(r['Promedio Global']) || 0), 0) / count;

    const overallColor = overall >= 4 ? 'var(--success)' : overall >= 2.5 ? 'var(--accent2)' : 'var(--danger)';


    const dimRows = DIMENSIONS.map(dim => {

      const avg = records.reduce((a,r) => a + (parseFloat(r[`Promedio ${dim.label}`]) || 0), 0) / count;

      const pct = (avg / 5) * 100;

      return `

        <div class="dim-bar-row">

          <div class="dim-bar-label">

            <span>${dim.label}</span>

            <span>${avg.toFixed(1)}</span>

          </div>

          <div class="dim-bar-track">

            <div class="dim-bar-fill" style="width:${pct}%;background:${dim.color}"></div>

          </div>

        </div>

      `;

    }).join('');


    return `

      <div class="leader-card">

        <h3>${leader}</h3>

        <p class="eval-count">${count} evaluación${count !== 1 ? 'es' : ''}</p>

        ${dimRows}

        <div class="overall-score">

          <span>Promedio Global</span>

          <span class="overall-num" style="color:${overallColor}">${overall.toFixed(2)}<span style="font-size:0.7rem;color:var(--muted)"> /5</span></span>

        </div>

      </div>

    `;

  }).join('');

}


// ─── UI HELPERS ───────────────────────────────────────────

function showTab(tab, btn) {

  document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));

  document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));

  document.getElementById(`tab-${tab}`).classList.add('active');

  btn.classList.add('active');

  if (tab === 'data') loadData();

  if (tab === 'compare') loadData();

}


function showToast(msg, isError = false) {

  const t = document.getElementById('toast');

  t.textContent = msg;

  t.className = 'toast show' + (isError ? ' error' : '');

  setTimeout(() => t.className = 'toast', 3000);

}
